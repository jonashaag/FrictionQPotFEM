97template <class T, class R>
98inline void
99VectorPartitionedTyings::asDofs_i(const T& nodevec, R& dofval_i, bool apply_tyings) const
100{
101 GOOSEFEM_ASSERT(xt::has_shape(nodevec, {m_nnode, m_ndim}));
102 GOOSEFEM_ASSERT(dofval_i.size() == m_nni);
103
104 dofval_i.fill(0.0);
105
106#pragma omp parallel for
107 for (size_t m = 0; m < m_nnode; ++m) {
108 for (size_t i = 0; i < m_ndim; ++i) {
109 if (m_dofs(m, i) < m_nni) {
110 dofval_i(m_dofs(m, i)) = nodevec(m, i);
111 }
112 }
113 }
114
115 if (!apply_tyings) {
116 return;
117 }
118
119 Eigen::VectorXd Dofval_d = this->Eigen_asDofs_d(nodevec);
120 Eigen::VectorXd Dofval_i = m_Cid * Dofval_d;
121
122#pragma omp parallel for
123 for (size_t i = 0; i < m_nni; ++i) {
124 dofval_i(i) += Dofval_i(i);
125 }
126}

136template <class T>
137inline Eigen::VectorXd VectorPartitionedTyings::Eigen_asDofs_d(const T& nodevec) const
138{
139 GOOSEFEM_ASSERT(xt::has_shape(nodevec, {m_nnode, m_ndim}));
140

142
143#pragma omp parallel for
