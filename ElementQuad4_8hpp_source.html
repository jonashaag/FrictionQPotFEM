144template <class T, class X, class W>
145inline Quadrature::Quadrature(const T& x, const X& xi, const W& w)
146{
147 m_x = x;
148 m_w = w;
149 m_xi = xi;
150 m_nip = w.size();
151 m_nelem = m_x.shape(0);
152 m_N = xt::empty<double>({m_nip, s_nne});
153 m_dNxi = xt::empty<double>({m_nip, s_nne, s_ndim});
154
155 for (size_t q = 0; q < m_nip; ++q) {
156 m_N(q, 0) = 0.25 * (1.0 - xi(q, 0)) * (1.0 - xi(q, 1));
157 m_N(q, 1) = 0.25 * (1.0 + xi(q, 0)) * (1.0 - xi(q, 1));
158 m_N(q, 2) = 0.25 * (1.0 + xi(q, 0)) * (1.0 + xi(q, 1));
159 m_N(q, 3) = 0.25 * (1.0 - xi(q, 0)) * (1.0 + xi(q, 1));
160 }
161
162 for (size_t q = 0; q < m_nip; ++q) {
163 // - dN / dxi_0
164 m_dNxi(q, 0, 0) = -0.25 * (1.0 - xi(q, 1));
165 m_dNxi(q, 1, 0) = +0.25 * (1.0 - xi(q, 1));
166 m_dNxi(q, 2, 0) = +0.25 * (1.0 + xi(q, 1));
167 m_dNxi(q, 3, 0) = -0.25 * (1.0 + xi(q, 1));
168 // - dN / dxi_1
169 m_dNxi(q, 0, 1) = -0.25 * (1.0 - xi(q, 0));
170 m_dNxi(q, 1, 1) = -0.25 * (1.0 + xi(q, 0));
171 m_dNxi(q, 2, 1) = +0.25 * (1.0 + xi(q, 0));
172 m_dNxi(q, 3, 1) = +0.25 * (1.0 - xi(q, 0));
173 }
174
175 GOOSEFEM_ASSERT(m_x.shape(1) == s_nne);
176 GOOSEFEM_ASSERT(m_x.shape(2) == s_ndim);
177 GOOSEFEM_ASSERT(xt::has_shape(m_xi, {m_nip, s_ndim}));
178 GOOSEFEM_ASSERT(xt::has_shape(m_w, {m_nip}));
179 GOOSEFEM_ASSERT(xt::has_shape(m_N, {m_nip, s_nne}));
180 GOOSEFEM_ASSERT(xt::has_shape(m_dNxi, {m_nip, s_nne, s_ndim}));
181
182 m_dNx = xt::empty<double>({m_nelem, m_nip, s_nne, s_ndim});
183 m_vol = xt::empty<double>(this->shape_qscalar());
184
185 this->compute_dN_impl();
186}

188inline void Quadrature::compute_dN_impl()
189{
190#pragma omp parallel
191 {
192 xt::xtensor<double, 2> J = xt::empty<double>({2, 2});
193 xt::xtensor<double, 2> Jinv = xt::empty<double>({2, 2});
194
195#pragma omp for
196 for (size_t e = 0; e < m_nelem; ++e) {
197
198 auto x = xt::adapt(&m_x(e, 0, 0), xt::xshape<s_nne, s_ndim>());
199
200 for (size_t q = 0; q < m_nip; ++q) {
201
202 auto dNxi = xt::adapt(&m_dNxi(q, 0, 0), xt::xshape<s_nne, s_ndim>());
203 auto dNx = xt::adapt(&m_dNx(e, q, 0, 0), xt::xshape<s_nne, s_ndim>());
204
205 // J(i,j) += dNxi(m,i) * x(m,j);
206 J(0, 0) = dNxi(0, 0) * x(0, 0) + dNxi(1, 0) * x(1, 0) + dNxi(2, 0) * x(2, 0) +
207 dNxi(3, 0) * x(3, 0);
208 J(0, 1) = dNxi(0, 0) * x(0, 1) + dNxi(1, 0) * x(1, 1) + dNxi(2, 0) * x(2, 1) +
209 dNxi(3, 0) * x(3, 1);
210 J(1, 0) = dNxi(0, 1) * x(0, 0) + dNxi(1, 1) * x(1, 0) + dNxi(2, 1) * x(2, 0) +
211 dNxi(3, 1) * x(3, 0);
212 J(1, 1) = dNxi(0, 1) * x(0, 1) + dNxi(1, 1) * x(1, 1) + dNxi(2, 1) * x(2, 1) +
213 dNxi(3, 1) * x(3, 1);
214
215 double Jdet = detail::tensor<2>::inv(J, Jinv);
216
217 // dNx(m,i) += Jinv(i,j) * dNxi(m,j);
218 for (size_t m = 0; m < s_nne; ++m) {
219 dNx(m, 0) = Jinv(0, 0) * dNxi(m, 0) + Jinv(0, 1) * dNxi(m, 1);
220 dNx(m, 1) = Jinv(1, 0) * dNxi(m, 0) + Jinv(1, 1) * dNxi(m, 1);
221 }
222
223 m_vol(e, q) = m_w(q) * Jdet;
224 }
225 }
226 }

341template <class T, class R>
342inline void Quadrature::int_N_scalar_NT_dV_impl(const T& qscalar, R& elemmat) const
343{
344 GOOSEFEM_ASSERT(xt::has_shape(qscalar, this->shape_qscalar()));
345 GOOSEFEM_ASSERT(xt::has_shape(elemmat, this->shape_elemmat()));
346
347 elemmat.fill(0.0);
348
349#pragma omp parallel for
350 for (size_t e = 0; e < m_nelem; ++e) {
351
352 auto M = xt::adapt(&elemmat(e, 0, 0), xt::xshape<s_nne * s_ndim, s_nne * s_ndim>());
353
354 for (size_t q = 0; q < m_nip; ++q) {
355
356 auto N = xt::adapt(&m_N(q, 0), xt::xshape<s_nne>());
357 auto& vol = m_vol(e, q);
358 auto& rho = qscalar(e, q);
359
360 // M(m*ndim+i,n*ndim+i) += N(m) * scalar * N(n) * dV
361 for (size_t m = 0; m < s_nne; ++m) {
362 for (size_t n = 0; n < s_nne; ++n) {
363 M(m * s_ndim + 0, n * s_ndim + 0) += N(m) * rho * N(n) * vol;
364 M(m * s_ndim + 1, n * s_ndim + 1) += N(m) * rho * N(n) * vol;
365 }
366 }
367 }
