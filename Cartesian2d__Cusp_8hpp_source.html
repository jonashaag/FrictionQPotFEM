24inline void Cusp::reset_epsy(const T& epsy, bool init_elastic)
25{
26 GMATELASTOPLASTICQPOT_ASSERT(epsy.size() > 0);
27 // assertion on epsy being sorted is done by QPot::Chunked
28
29 if (!init_elastic) {
30 m_yield = QPot::Chunked(0.0, epsy, 0);
31 return;

33
34 GMATELASTOPLASTICQPOT_ASSERT(epsy.front() > 0);
35 std::vector<double> y(epsy.size() + 1);
36 y[0] = -epsy.front();

38 m_yield = QPot::Chunked(0.0, y, 0);
39}

91inline double Cusp::energy() const
92{
93 namespace GT = GMatTensor::Cartesian2d::pointer;
94 std::array<double, 4> Epsd;
95 double epsm = GT::Hydrostatic_deviatoric(&m_Eps[0], &Epsd[0]);
96 double epsd = std::sqrt(0.5 * GT::A2s_ddot_B2s(&Epsd[0], &Epsd[0]));
97 double U = m_K * std::pow(epsm, 2.0);
98
99 double eps_min = 0.5 * (m_yield.yright() + m_yield.yleft());
100 double deps_y = 0.5 * (m_yield.yright() - m_yield.yleft());
101
102 double V = m_G * (std::pow(epsd - eps_min, 2.0) - std::pow(deps_y, 2.0));
103
104 return U + V;
105}

122template <class T>
123inline void Cusp::setStrainPtr(const T* arg)
124{
125 namespace GT = GMatTensor::Cartesian2d::pointer;
126 std::copy(arg, arg + 4, m_Eps.begin());
127
128 std::array<double, 4> Epsd;
129 double epsm = GT::Hydrostatic_deviatoric(&m_Eps[0], &Epsd[0]);
130 double epsd = std::sqrt(0.5 * GT::A2s_ddot_B2s(&Epsd[0], &Epsd[0]));
131 m_yield.set_x(epsd);
132
133 m_Sig[0] = m_Sig[3] = m_K * epsm;
134
135 if (epsd <= 0.0) {
136 m_Sig[1] = m_Sig[2] = 0.0;
137 return;
138 }
139
140 double eps_min = 0.5 * (m_yield.yright() + m_yield.yleft());
141
142 double g = m_G * (1.0 - eps_min / epsd);
143 m_Sig[0] += g * Epsd[0];
144 m_Sig[1] = g * Epsd[1];
145 m_Sig[2] = g * Epsd[2];
146 m_Sig[3] += g * Epsd[3];
147}
