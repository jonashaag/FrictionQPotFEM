18template <class S, class T>
19inline ElementType defaultElementType(const S& coor, const T& conn)
20{
21 GOOSEFEM_ASSERT(coor.dimension() == 2);
22 GOOSEFEM_ASSERT(conn.dimension() == 2);
23
24 if (coor.shape(1) == 2ul && conn.shape(1) == 3ul) {
25 return ElementType::Tri3;
26 }
27 if (coor.shape(1) == 2ul && conn.shape(1) == 4ul) {
28 return ElementType::Quad4;
29 }
30 if (coor.shape(1) == 3ul && conn.shape(1) == 8ul) {
31 return ElementType::Hex8;
32 }
33
34 throw std::runtime_error("Element-type not implemented");
35}

239inline xt::xtensor<size_t, 2> RegularBase2d<D>::nodesPeriodic_impl() const
240{
241 xt::xtensor<size_t, 1> bot = derived_cast().nodesBottomOpenEdge_impl();
242 xt::xtensor<size_t, 1> top = derived_cast().nodesTopOpenEdge_impl();
243 xt::xtensor<size_t, 1> lft = derived_cast().nodesLeftOpenEdge_impl();
244 xt::xtensor<size_t, 1> rgt = derived_cast().nodesRightOpenEdge_impl();
245 std::array<size_t, 2> shape = {bot.size() + lft.size() + size_t(3), size_t(2)};
246 auto ret = xt::xtensor<size_t, 2>::from_shape(shape);
247
248 ret(0, 0) = derived_cast().nodesBottomLeftCorner_impl();
249 ret(0, 1) = derived_cast().nodesBottomRightCorner_impl();
250
251 ret(1, 0) = derived_cast().nodesBottomLeftCorner_impl();
252 ret(1, 1) = derived_cast().nodesTopRightCorner_impl();
253
254 ret(2, 0) = derived_cast().nodesBottomLeftCorner_impl();
255 ret(2, 1) = derived_cast().nodesTopLeftCorner_impl();
256
257 size_t i = 3;
258
259 xt::view(ret, xt::range(i, i + bot.size()), 0) = bot;
260 xt::view(ret, xt::range(i, i + bot.size()), 1) = top;
261
262 i += bot.size();
263
264 xt::view(ret, xt::range(i, i + lft.size()), 0) = lft;
265 xt::view(ret, xt::range(i, i + lft.size()), 1) = rgt;
266
267 return ret;
268}

1095template <class S, class T>
1096inline xt::xtensor<size_t, 2>
1097overlapping(const S& coor_a, const T& coor_b, double rtol, double atol)
1098{
1099 GOOSEFEM_ASSERT(coor_a.dimension() == 2);
1100 GOOSEFEM_ASSERT(coor_b.dimension() == 2);
1101 GOOSEFEM_ASSERT(coor_a.shape(1) == coor_b.shape(1));
1102
1103 std::vector<size_t> ret_a;
1104 std::vector<size_t> ret_b;
1105
1106 for (size_t i = 0; i < coor_a.shape(0); ++i) {
1107
1108 auto idx = xt::flatten_indices(xt::argwhere(
1109 xt::prod(xt::isclose(coor_b, xt::view(coor_a, i, xt::all()), rtol, atol), 1)));
1110
1111 for (auto& j : idx) {
1112 ret_a.push_back(i);
1113 ret_b.push_back(j);
1114 }
1115 }
1116
1117 xt::xtensor<size_t, 2> ret = xt::empty<size_t>({size_t(2), ret_a.size()});
1118 for (size_t i = 0; i < ret_a.size(); ++i) {
1119 ret(0, i) = ret_a[i];
1120 ret(1, i) = ret_b[i];
1121 }
1122
1123 return ret;
1124}
