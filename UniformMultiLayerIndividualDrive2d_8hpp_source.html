202template <class S, class T>
203inline double System::initEventDriven(const S& ubar, const T& active)
204{
205 FRICTIONQPOTFEM_ASSERT(xt::has_shape(ubar, m_layerdrive_targetubar.shape()));
206 FRICTIONQPOTFEM_ASSERT(xt::has_shape(active, m_layerdrive_active.shape()));
207
208 // backup system
209
210 auto u0 = m_u;
211 auto v0 = m_v;
212 auto a0 = m_a;
213 auto active0 = m_layerdrive_active;
214 auto ubar0 = m_layerdrive_targetubar;
215 auto t0 = m_t;
216 xt::xtensor<double, 3> epsy0(std::array<size_t, 3>{m_nelem_plas, m_nip, 2});
217
218 for (size_t e = 0; e < m_nelem_plas; ++e) {
219 for (size_t q = 0; q < m_nip; ++q) {
220 auto& cusp = m_material_plas.refCusp({e, q});
221 auto epsy = cusp.epsy();
222 epsy0(e, q, 0) = epsy(0);
223 epsy0(e, q, 1) = epsy(1);
224 epsy(1) = std::numeric_limits<double>::max();
225 epsy(0) = -epsy(1);
226 cusp.reset_epsy(epsy, false);
227 }
228 }
229
230 // perturbation
231
232 m_u.fill(0.0);
233 m_v.fill(0.0);
234 m_a.fill(0.0);
235 this->updated_u();
236 FRICTIONQPOTFEM_ASSERT(xt::all(xt::equal(this->plastic_CurrentIndex(), 0)));
237 this->layerSetTargetActive(active);
238 this->layerSetTargetUbar(ubar);
239 this->minimise();
240 FRICTIONQPOTFEM_ASSERT(xt::all(xt::equal(this->plastic_CurrentIndex(), 0)));
241
242 auto c = this->eventDriven_setDeltaU(m_u);
243 m_pert_layerdrive_active = active;
244 m_pert_layerdrive_targetubar = c * ubar;
245
246 // restore system
247
248 for (size_t e = 0; e < m_nelem_plas; ++e) {
249 for (size_t q = 0; q < m_nip; ++q) {
250 auto& cusp = m_material_plas.refCusp({e, q});
251 auto epsy = cusp.epsy();
252 epsy(0) = epsy0(e, q, 0);
253 epsy(1) = epsy0(e, q, 1);
254 cusp.reset_epsy(epsy, false);
255 }
256 }
257
258 this->setU(u0);
259 this->setV(v0);
260 this->setA(a0);
261 this->layerSetTargetActive(active0);
262 this->layerSetTargetUbar(ubar0);
263 this->setT(t0);
264
265 return c;
266}

443inline void System::computeForceFromTargetUbar()
444{
445 m_uq.fill(0.0); // pre-allocated value that an be freely used
446 size_t nip = m_quad.nip();
447
448 for (size_t i = 0; i < m_n_layer; ++i) {
449 for (size_t d = 0; d < 2; ++d) {
450 if (m_layerdrive_active(i, d)) {
451 double f = m_drive_k * (m_layer_ubar(i, d) - m_layerdrive_targetubar(i, d));
452 if (m_drive_spring_symmetric || f < 0) { // buckle under compression
453 for (auto& e : m_layer_elem[i]) {
454 for (size_t q = 0; q < nip; ++q) {
455 m_uq(e, q, d) = f;
456 }
457 }
458 }
459 }
460 }
461 }
462
463 m_quad.int_N_vector_dV(m_uq, m_ue);
464 m_vector.assembleDofs(m_ue, m_ud);
465 m_vector.asNode(m_ud, m_fdrive);
466}
