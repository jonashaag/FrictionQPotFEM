105inline void MatrixPartitionedTyings::assemble(const xt::xtensor<double, 3>& elemmat)
106{
107 GOOSEFEM_ASSERT(xt::has_shape(elemmat, {m_nelem, m_nne * m_ndim, m_nne * m_ndim}));
108
109 m_Tuu.clear();
110 m_Tup.clear();
111 m_Tpu.clear();
112 m_Tpp.clear();
113 m_Tud.clear();
114 m_Tpd.clear();
115 m_Tdu.clear();
116 m_Tdp.clear();

119 for (size_t e = 0; e < m_nelem; ++e) {
120 for (size_t m = 0; m < m_nne; ++m) {
121 for (size_t i = 0; i < m_ndim; ++i) {
122
123 size_t di = m_dofs(m_conn(e, m), i);

125 for (size_t n = 0; n < m_nne; ++n) {

133 }
134 else if (di < m_nnu && dj < m_nni) {
135 m_Tup.push_back(Eigen::Triplet<double>(
136 di, dj - m_nnu, elemmat(e, m * m_ndim + i, n * m_ndim + j)));
137 }
138 else if (di < m_nnu) {
139 m_Tud.push_back(Eigen::Triplet<double>(
140 di, dj - m_nni, elemmat(e, m * m_ndim + i, n * m_ndim + j)));
141 }
142 else if (di < m_nni && dj < m_nnu) {
143 m_Tpu.push_back(Eigen::Triplet<double>(
144 di - m_nnu, dj, elemmat(e, m * m_ndim + i, n * m_ndim + j)));
145 }
146 else if (di < m_nni && dj < m_nni) {
147 m_Tpp.push_back(Eigen::Triplet<double>(
148 di - m_nnu,
149 dj - m_nnu,
150 elemmat(e, m * m_ndim + i, n * m_ndim + j)));
151 }
152 else if (di < m_nni) {
153 m_Tpd.push_back(Eigen::Triplet<double>(
154 di - m_nnu,
155 dj - m_nni,
156 elemmat(e, m * m_ndim + i, n * m_ndim + j)));
157 }
158 else if (dj < m_nnu) {
159 m_Tdu.push_back(Eigen::Triplet<double>(
160 di - m_nni, dj, elemmat(e, m * m_ndim + i, n * m_ndim + j)));
161 }
162 else if (dj < m_nni) {
163 m_Tdp.push_back(Eigen::Triplet<double>(
164 di - m_nni,
165 dj - m_nnu,
166 elemmat(e, m * m_ndim + i, n * m_ndim + j)));
167 }
168 else {
169 m_Tdd.push_back(Eigen::Triplet<double>(
170 di - m_nni,
171 dj - m_nni,
172 elemmat(e, m * m_ndim + i, n * m_ndim + j)));
173 }
174 }
175 }
176 }
177 }
178 }

293{
294 if (!matrix.m_changed && !m_factor) {
295 return;
296 }
297
298 matrix.m_ACuu = matrix.m_Auu + matrix.m_Aud * matrix.m_Cdu + matrix.m_Cud * matrix.m_Adu +

315template <class Solver>
316inline void MatrixPartitionedTyingsSolver<Solver>::solve(
317 MatrixPartitionedTyings& matrix,
318 const xt::xtensor<double, 2>& b,
319 xt::xtensor<double, 2>& x)
320{
321 GOOSEFEM_ASSERT(xt::has_shape(b, {matrix.m_nnode, matrix.m_ndim}));
322 GOOSEFEM_ASSERT(xt::has_shape(x, {matrix.m_nnode, matrix.m_ndim}));
323
324 this->factorize(matrix);
325
326 Eigen::VectorXd B_u = matrix.AsDofs_u(b);
327 Eigen::VectorXd B_d = matrix.AsDofs_d(b);
328 Eigen::VectorXd X_p = matrix.AsDofs_p(x);
329
330 B_u += matrix.m_Cud * B_d;
331
332 Eigen::VectorXd X_u = m_solver.solve(Eigen::VectorXd(B_u - matrix.m_ACup * X_p));
333 Eigen::VectorXd X_d = matrix.m_Cdu * X_u + matrix.m_Cdp * X_p;
334
335#pragma omp parallel for
336 for (size_t m = 0; m < matrix.m_nnode; ++m) {
337 for (size_t i = 0; i < matrix.m_ndim; ++i) {
338 if (matrix.m_dofs(m, i) < matrix.m_nnu) {
339 x(m, i) = X_u(matrix.m_dofs(m, i));
340 }
341 else if (matrix.m_dofs(m, i) >= matrix.m_nni) {
342 x(m, i) = X_d(matrix.m_dofs(m, i) - matrix.m_nni);
343 }
344 }
345 }

349inline void MatrixPartitionedTyingsSolver<Solver>::solve(
350 MatrixPartitionedTyings& matrix,
351 const xt::xtensor<double, 1>& b,
352 xt::xtensor<double, 1>& x)
353{
354 GOOSEFEM_ASSERT(b.size() == matrix.m_ndof);
355 GOOSEFEM_ASSERT(x.size() == matrix.m_ndof);
356
357 this->factorize(matrix);
358
359 Eigen::VectorXd B_u = matrix.AsDofs_u(b);
360 Eigen::VectorXd B_d = matrix.AsDofs_d(b);
361 Eigen::VectorXd X_p = matrix.AsDofs_p(x);
362
363 Eigen::VectorXd X_u = m_solver.solve(Eigen::VectorXd(B_u - matrix.m_ACup * X_p));
364 Eigen::VectorXd X_d = matrix.m_Cdu * X_u + matrix.m_Cdp * X_p;
365
366#pragma omp parallel for
367 for (size_t d = 0; d < matrix.m_nnu; ++d) {
368 x(matrix.m_iiu(d)) = X_u(d);
369 }
370
371#pragma omp parallel for
372 for (size_t d = 0; d < matrix.m_nnd; ++d) {
373 x(matrix.m_iid(d)) = X_d(d);
374 }
