61inline void MatrixDiagonal::factorize()
62{
63 if (!m_changed) {
64 return;
65 }
66
67#pragma omp parallel for
68 for (size_t d = 0; d < m_ndof; ++d) {
69 m_inv(d) = 1.0 / m_A(d);

82inline void MatrixDiagonal::assemble(const xt::xtensor<double, 3>& elemmat)
83{
84 GOOSEFEM_ASSERT(xt::has_shape(elemmat, {m_nelem, m_nne * m_ndim, m_nne * m_ndim}));
85 GOOSEFEM_ASSERT(Element::isDiagonal(elemmat));
86
87 m_A.fill(0.0);
88
89 for (size_t e = 0; e < m_nelem; ++e) {
90 for (size_t m = 0; m < m_nne; ++m) {
91 for (size_t i = 0; i < m_ndim; ++i) {
92 m_A(m_dofs(m_conn(e, m), i)) += elemmat(e, m * m_ndim + i, m * m_ndim + i);
93 }
94 }
95 }
96
97 m_changed = true;
98}
