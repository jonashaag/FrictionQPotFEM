30inline Periodic::Periodic(
31 const xt::xtensor<double, 2>& coor,
32 const xt::xtensor<size_t, 2>& dofs,
33 const xt::xtensor<size_t, 2>& control,
34 const xt::xtensor<size_t, 2>& nodal_tyings,
35 const xt::xtensor<size_t, 1>& iip)
36 : m_tyings(nodal_tyings), m_coor(coor)
37{
38 m_ndim = m_coor.shape(1);
39 m_nties = m_tyings.shape(0);
40
41 GOOSEFEM_ASSERT(xt::has_shape(m_tyings, {m_nties, size_t(2)}));
42 GOOSEFEM_ASSERT(xt::has_shape(control, {m_ndim, m_ndim}));
43 GOOSEFEM_ASSERT(xt::has_shape(dofs, m_coor.shape()));
44 GOOSEFEM_ASSERT(xt::amax(control)() <= xt::amax(dofs)());
45 GOOSEFEM_ASSERT(xt::amin(control)() >= xt::amin(dofs)());
46 GOOSEFEM_ASSERT(xt::amax(iip)() <= xt::amax(dofs)());
47 GOOSEFEM_ASSERT(xt::amin(iip)() >= xt::amin(dofs)());
48 GOOSEFEM_ASSERT(xt::amax(nodal_tyings)() < m_coor.shape(0));
49
50 xt::xtensor<size_t, 1> dependent = xt::view(m_tyings, xt::all(), 1);
51 xt::xtensor<size_t, 2> dependent_dofs = xt::view(dofs, xt::keep(dependent), xt::all());
52 xt::xtensor<size_t, 1> iid = xt::flatten(dependent_dofs);
53 xt::xtensor<size_t, 1> iii = xt::setdiff1d(dofs, iid);
54 xt::xtensor<size_t, 1> iiu = xt::setdiff1d(iii, iip);
55
56 m_nnu = iiu.size();
57 m_nnp = iip.size();
58 m_nni = iii.size();
59 m_nnd = iid.size();
60
61 GooseFEM::Mesh::Reorder reorder({iiu, iip, iid});
62
63 m_dofs = reorder.apply(dofs);
64 m_control = reorder.apply(control);
65}

122inline Eigen::SparseMatrix<double> Periodic::Cdi() const
123{
124 std::vector<Eigen::Triplet<double>> data;
125
126 data.reserve(m_nties * m_ndim * (m_ndim + 1));
127
128 for (size_t i = 0; i < m_nties; ++i) {
129 for (size_t j = 0; j < m_ndim; ++j) {
130
131 size_t ni = m_tyings(i, 0);
132 size_t nd = m_tyings(i, 1);
133
134 data.push_back(Eigen::Triplet<double>(i * m_ndim + j, m_dofs(ni, j), +1.0));
135
136 for (size_t k = 0; k < m_ndim; ++k) {
137 data.push_back(Eigen::Triplet<double>(
138 i * m_ndim + j, m_control(j, k), m_coor(nd, k) - m_coor(ni, k)));
139 }
140 }
141 }
142
143 Eigen::SparseMatrix<double> Cdi;
144 Cdi.resize(m_nnd, m_nni);
145 Cdi.setFromTriplets(data.begin(), data.end());
146
147 return Cdi;
148}

214inline Control::Control(const xt::xtensor<double, 2>& coor, const xt::xtensor<size_t, 2>& dofs)
215 : m_coor(coor), m_dofs(dofs)
216{
217 GOOSEFEM_ASSERT(coor.shape().size() == 2);
218 GOOSEFEM_ASSERT(coor.shape() == dofs.shape());
219
220 size_t nnode = coor.shape(0);
221 size_t ndim = coor.shape(1);
222
223 m_control_dofs = xt::arange<size_t>(ndim * ndim).reshape({ndim, ndim});
224 m_control_dofs += xt::amax(dofs)() + 1;
225
226 m_control_nodes = nnode + xt::arange<size_t>(ndim);
227
228 m_coor = xt::concatenate(xt::xtuple(coor, xt::zeros<double>({ndim, ndim})));
229 m_dofs = xt::concatenate(xt::xtuple(dofs, m_control_dofs));
230}
