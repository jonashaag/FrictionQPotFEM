73{

78 m_Tpu.clear();
79 m_Tpp.clear();

84

249inline void MatrixPartitioned::dot(const xt::xtensor<double, 2>& x, xt::xtensor<double, 2>& b) const
250{
251 GOOSEFEM_ASSERT(xt::has_shape(b, {m_nnode, m_ndim}));
252 GOOSEFEM_ASSERT(xt::has_shape(x, {m_nnode, m_ndim}));
253
254 Eigen::VectorXd X_u = this->AsDofs_u(x);
255 Eigen::VectorXd X_p = this->AsDofs_p(x);
256 Eigen::VectorXd B_u = m_Auu * X_u + m_Aup * X_p;
257 Eigen::VectorXd B_p = m_Apu * X_u + m_App * X_p;
258
259#pragma omp parallel for
260 for (size_t m = 0; m < m_nnode; ++m) {
261 for (size_t i = 0; i < m_ndim; ++i) {
262 if (m_part(m, i) < m_nnu) {
263 b(m, i) = B_u(m_part(m, i));
264 }
265 else {
266 b(m, i) = B_p(m_part(m, i) - m_nnu);
267 }
268 }
269 }

449{

462 x(m, i) = X_u(matrix.m_part(m, i));
