253inline void Static::setPosition(double x)
254{
255 QPOT_ASSERT(x > m_min);
256 QPOT_ASSERT(x < m_max);
257
258 if (m_left < x && m_right >= x) {
259 return;
260 }
261
262 size_t l = m_idx > m_proximity ? m_idx - m_proximity : 0;
263 size_t r = std::min(m_idx + m_proximity, m_ntot - 1);
264
265 if (m_pos(l) < x && m_pos(r) >= x) {
266 m_idx = std::lower_bound(&m_pos(l), &m_pos(l) + r - l, x) - &m_pos(l) - 1 + l;
267 }
268 else {
269 m_idx = std::lower_bound(m_pos.begin(), m_pos.end(), x) - m_pos.begin() - 1;
270 }
271
272 m_left = m_pos(m_idx);
273 m_right = m_pos(m_idx + 1);
