163inline Quadrature::Quadrature(const T& x, const X& xi, const W& w)
164{

166 m_w = w;
167 m_xi = xi;
168 m_nip = w.size();
169 m_nelem = m_x.shape(0);

172
173 for (size_t q = 0; q < m_nip; ++q) {
174 m_N(q, 0) = 0.125 * (1.0 - xi(q, 0)) * (1.0 - xi(q, 1)) * (1.0 - xi(q, 2));
175 m_N(q, 1) = 0.125 * (1.0 + xi(q, 0)) * (1.0 - xi(q, 1)) * (1.0 - xi(q, 2));
176 m_N(q, 2) = 0.125 * (1.0 + xi(q, 0)) * (1.0 + xi(q, 1)) * (1.0 - xi(q, 2));
177 m_N(q, 3) = 0.125 * (1.0 - xi(q, 0)) * (1.0 + xi(q, 1)) * (1.0 - xi(q, 2));
178 m_N(q, 4) = 0.125 * (1.0 - xi(q, 0)) * (1.0 - xi(q, 1)) * (1.0 + xi(q, 2));
179 m_N(q, 5) = 0.125 * (1.0 + xi(q, 0)) * (1.0 - xi(q, 1)) * (1.0 + xi(q, 2));
180 m_N(q, 6) = 0.125 * (1.0 + xi(q, 0)) * (1.0 + xi(q, 1)) * (1.0 + xi(q, 2));
181 m_N(q, 7) = 0.125 * (1.0 - xi(q, 0)) * (1.0 + xi(q, 1)) * (1.0 + xi(q, 2));
182 }

227template <class T, class R>
228inline void Quadrature::int_N_scalar_NT_dV_impl(const T& qscalar, R& elemmat) const
229{
230 GOOSEFEM_ASSERT(xt::has_shape(qscalar, this->shape_qscalar()));
231 GOOSEFEM_ASSERT(xt::has_shape(elemmat, this->shape_elemmat()));
232
233 elemmat.fill(0.0);
234
235#pragma omp parallel for
236 for (size_t e = 0; e < m_nelem; ++e) {
237
238 auto M = xt::adapt(&elemmat(e, 0, 0), xt::xshape<s_nne * s_ndim, s_nne * s_ndim>());
239
240 for (size_t q = 0; q < m_nip; ++q) {
241
242 auto N = xt::adapt(&m_N(q, 0), xt::xshape<s_nne>());
243 auto& vol = m_vol(e, q);
244 auto& rho = qscalar(e, q);
245
246 // M(m * ndim + i, n * ndim + i) += N(m) * scalar * N(n) * dV
247 for (size_t m = 0; m < s_nne; ++m) {
248 for (size_t n = 0; n < s_nne; ++n) {
249 M(m * s_ndim + 0, n * s_ndim + 0) += N(m) * rho * N(n) * vol;
250 M(m * s_ndim + 1, n * s_ndim + 1) += N(m) * rho * N(n) * vol;
251 M(m * s_ndim + 2, n * s_ndim + 2) += N(m) * rho * N(n) * vol;
252 }
253 }
254 }
255 }

259inline void Quadrature::int_gradN_dot_tensor2_dV_impl(const T& qtensor, R& elemvec) const
260{
261 GOOSEFEM_ASSERT(xt::has_shape(qtensor, this->shape_qtensor<2>()));
262 GOOSEFEM_ASSERT(xt::has_shape(elemvec, this->shape_elemvec()));
263
264 elemvec.fill(0.0);
265
266#pragma omp parallel for
267 for (size_t e = 0; e < m_nelem; ++e) {
268
269 auto f = xt::adapt(&elemvec(e, 0, 0), xt::xshape<s_nne, s_ndim>());
270
271 for (size_t q = 0; q < m_nip; ++q) {
272
273 auto dNx = xt::adapt(&m_dNx(e, q, 0, 0), xt::xshape<s_nne, s_ndim>());
274 auto sig = xt::adapt(&qtensor(e, q, 0, 0), xt::xshape<s_ndim, s_ndim>());
275 auto& vol = m_vol(e, q);
276
277 for (size_t m = 0; m < s_nne; ++m) {
278 f(m, 0) +=
279 (dNx(m, 0) * sig(0, 0) + dNx(m, 1) * sig(1, 0) + dNx(m, 2) * sig(2, 0)) * vol;
280 f(m, 1) +=
281 (dNx(m, 0) * sig(0, 1) + dNx(m, 1) * sig(1, 1) + dNx(m, 2) * sig(2, 1)) * vol;
282 f(m, 2) +=
283 (dNx(m, 0) * sig(0, 2) + dNx(m, 1) * sig(1, 2) + dNx(m, 2) * sig(2, 2)) * vol;
284 }
