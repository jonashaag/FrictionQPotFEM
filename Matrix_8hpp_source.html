62inline void Matrix::assemble(const xt::xtensor<double, 3>& elemmat)
63{
64 GOOSEFEM_ASSERT(xt::has_shape(elemmat, {m_nelem, m_nne * m_ndim, m_nne * m_ndim}));
65
66 m_T.clear();
67
68 for (size_t e = 0; e < m_nelem; ++e) {
69 for (size_t m = 0; m < m_nne; ++m) {
70 for (size_t i = 0; i < m_ndim; ++i) {
71 for (size_t n = 0; n < m_nne; ++n) {
72 for (size_t j = 0; j < m_ndim; ++j) {
73 m_T.push_back(Eigen::Triplet<double>(
74 m_dofs(m_conn(e, m), i),
75 m_dofs(m_conn(e, n), j),
76 elemmat(e, m * m_ndim + i, n * m_ndim + j)));
77 }
78 }
79 }
80 }
81 }
82
83 m_A.setFromTriplets(m_T.begin(), m_T.end());
84 m_changed = true;
85}
