149inline double System::addSimpleShearToFixedStress(double target_stress, bool dry_run)
150{
151 FRICTIONQPOTFEM_ASSERT(this->isHomogeneousElastic());
152
153 auto u_new = this->u();
154 auto dV = m_quad.AsTensor<2>(this->dV());
155 double G = m_material.G().data()[0];
156
157 xt::xtensor<double, 2> Epsbar = xt::average(this->Eps(), dV, {0, 1});
158 xt::xtensor<double, 2> Sigbar = xt::average(this->Sig(), dV, {0, 1});
159 xt::xtensor<double, 2> Epsd = GMatTensor::Cartesian2d::Deviatoric(Epsbar);
160 double epsxx = Epsd(0, 0);
161 double epsxy = Epsd(0, 1);
162
163 FRICTIONQPOTFEM_ASSERT(Sigbar(0, 1) >= 0);
164
165 double eps = GMatElastoPlasticQPot::Cartesian2d::Epsd(Epsbar)();
166 double sig = GMatElastoPlasticQPot::Cartesian2d::Sigd(Sigbar)();
167 double direction = +1.0;
168 if (target_stress < sig) {
169 direction = -1.0;
170 }
171
172 double eps_new = eps + (target_stress - sig) / (2.0 * G);
173 double dgamma =
174 2.0 * (-1.0 * direction * epsxy + std::sqrt(std::pow(eps_new, 2.0) - std::pow(epsxx, 2.0)));
175
176 if (dry_run) {
177 return direction * dgamma;
178 }
179
180 for (size_t n = 0; n < m_nnode; ++n) {
181 u_new(n, 0) += direction * dgamma * (m_coor(n, 1) - m_coor(0, 1));
182 }
183 this->setU(u_new);
184
185 // sanity check
186 // ------------
187
188 Sigbar = xt::average(this->Sig(), dV, {0, 1});
189 sig = GMatElastoPlasticQPot::Cartesian2d::Sigd(Sigbar)();
190
191 FRICTIONQPOTFEM_REQUIRE(std::abs(target_stress - sig) / sig < 1e-4);
192
193 return direction * dgamma;
194}

204inline double System::triggerElementWithLocalSimpleShear(
205 double deps_kick,
206 size_t plastic_element,
207 bool trigger_weakest,
208 double amplify)
209{
210 FRICTIONQPOTFEM_ASSERT(plastic_element < m_nelem_plas);
211
212 auto idx = this->plastic_CurrentIndex();
213 auto eps = GMatElastoPlasticQPot::Cartesian2d::Epsd(this->plastic_Eps());
214 auto up = m_vector.AsDofs_p(m_u);
215
216 // distance to yielding on the positive side
217 auto epsy = this->plastic_CurrentYieldRight();
218 auto deps = epsy - eps;
219
220 // find integration point closest to yielding
221 size_t e = m_elem_plas(plastic_element);
222 size_t q = xt::argmin(xt::view(deps, plastic_element, xt::all()))();
223 if (!trigger_weakest) {
224 q = xt::argmax(xt::view(deps, plastic_element, xt::all()))();
225 }
226
227 // deviatoric strain at the selected quadrature-point
228 xt::xtensor<double, 2> Eps = xt::view(this->plastic_Eps(), plastic_element, q);
229 xt::xtensor<double, 2> Epsd = GMatTensor::Cartesian2d::Deviatoric(Eps);
230
231 // new equivalent deviatoric strain: yield strain + small strain kick
232 double eps_new = epsy(plastic_element, q) + deps_kick / 2.0;
233
234 // convert to increment in shear strain (N.B. "dgamma = 2 * Epsd(0,1)")
235 double dgamma =
236 2.0 * (-Epsd(0, 1) + std::sqrt(std::pow(eps_new, 2.0) - std::pow(Epsd(0, 0), 2.0)));
237
238 // apply increment in shear strain as a perturbation to the selected element
239 // - nodes belonging to the current element, from connectivity
240 auto elem = xt::view(m_conn, e, xt::all());
241 // - displacement-DOFs
242 auto udofs = m_vector.AsDofs(m_u);
243 // - update displacement-DOFs for the element
244 for (size_t n = 0; n < m_nne; ++n) {
245 udofs(m_dofs(elem(n), 0)) += dgamma * amplify * (m_coor(elem(n), 1) - m_coor(elem(0), 1));
246 }
247 // - convert displacement-DOFs to (periodic) nodal displacement vector
248 // (N.B. storing to nodes directly does not ensure periodicity)
249 this->setU(m_vector.AsNode(udofs));
250
251 eps = GMatElastoPlasticQPot::Cartesian2d::Epsd(this->plastic_Eps());
252 auto idx_new = this->plastic_CurrentIndex();
253 auto up_new = m_vector.AsDofs_p(m_u);
254
255 FRICTIONQPOTFEM_REQUIRE(dgamma >= 0.0);
256 FRICTIONQPOTFEM_REQUIRE(amplify >= 0.0);
257 FRICTIONQPOTFEM_REQUIRE(
258 std::abs(eps(plastic_element, q) - eps_new) / eps_new < 1e-4 || amplify != 1);
259 FRICTIONQPOTFEM_REQUIRE(xt::any(xt::not_equal(idx, idx_new)));
260 FRICTIONQPOTFEM_REQUIRE(idx(plastic_element, q) != idx_new(plastic_element, q));
261 FRICTIONQPOTFEM_REQUIRE(xt::allclose(up, up_new));
262
263 return dgamma;
264}

266inline xt::xtensor<double, 2>
267System::plastic_ElementYieldBarrierForSimpleShear(double deps_kick, size_t iquad)
268{
269 FRICTIONQPOTFEM_ASSERT(iquad < m_nip);
270
271 auto eps = GMatElastoPlasticQPot::Cartesian2d::Epsd(this->plastic_Eps());
272 auto epsy = this->plastic_CurrentYieldRight();
273 auto deps = xt::eval(epsy - eps);
274 xt::xtensor<double, 2> ret = xt::empty<double>({m_N, size_t(2)});
275 auto isort = xt::argsort(deps, 1);
276
277 auto Eps = this->plastic_Eps();
278 auto Epsd = GMatTensor::Cartesian2d::Deviatoric(Eps);
279
280 for (size_t e = 0; e < m_N; ++e) {
281 size_t q = isort(e, iquad);
282 double eps_new = epsy(e, q) + deps_kick / 2.0;
283 double gamma = Epsd(e, q, 0, 1);
284 double epsd_xx = Epsd(e, q, 0, 0);
285 ret(e, 0) = deps(e, q);
286 ret(e, 1) = -gamma + std::sqrt(std::pow(eps_new, 2.0) - std::pow(epsd_xx, 2.0));
287 }
288
289 return ret;
290}

292inline LocalTriggerFineLayerFull::LocalTriggerFineLayerFull(const System& sys)
293{
294 // Copy / allocate local variables
295
296 auto m = sys.material();
297 GMatElastoPlasticQPot::Cartesian2d::Array<2> material(m.shape());
298 material.setElastic(m.K(), m.G());
299
300 FRICTIONQPOTFEM_ASSERT(
301 xt::all(xt::equal(material.type(), GMatElastoPlasticQPot::Cartesian2d::Type::Elastic)));
302
303 auto vector = sys.vector();
304 auto K = sys.stiffness();
305 auto quad = sys.quad();
306 GooseFEM::MatrixPartitionedSolver<> solver;
307
308 m_elem_plas = sys.plastic();
309 m_nelem_plas = m_elem_plas.size();
310 m_nip = quad.nip();
311
312 m_smin = xt::zeros<double>({m_nelem_plas, m_nip});
313 m_pmin = xt::zeros<double>({m_nelem_plas, m_nip});
314 m_Wmin = xt::zeros<double>({m_nelem_plas, m_nip});
315 m_dgamma = xt::zeros<double>({m_nelem_plas, m_nip});
316 m_dE = xt::zeros<double>({m_nelem_plas, m_nip});
317
318 auto coor = sys.coor();
319 auto conn = sys.conn();
320
321 auto u = vector.allocate_nodevec(0.0);
322 auto fint = vector.allocate_nodevec(0.0);
323 auto fext = vector.allocate_nodevec(0.0);
324 auto fres = vector.allocate_nodevec(0.0);
325
326 auto ue = vector.allocate_elemvec(0.0);
327 auto fe = vector.allocate_elemvec(0.0);
328
329 auto Eps = quad.allocate_qtensor<2>(0.0);
330 auto Sig = quad.allocate_qtensor<2>(0.0);
331 std::copy(Eps.shape().cbegin(), Eps.shape().cend(), m_shape_T2.begin());
332
333 m_dV = quad.dV();
334 m_V = xt::sum(xt::view(m_dV, m_elem_plas(0)))();
335
336 // Replicate mesh
337
338 GooseFEM::Mesh::Quad4::FineLayer mesh(sys.coor(), sys.conn());
339
340 auto elmap = mesh.roll(1);
341 size_t nconfig = m_elem_plas(m_nelem_plas - 1) - elmap(m_elem_plas(m_nelem_plas - 1));
342 size_t nroll = m_nelem_plas / nconfig;
343
344 for (size_t roll = 0; roll < nroll; ++roll) {
345 m_elemmap.push_back(mesh.roll(roll));
346 m_nodemap.push_back(GooseFEM::Mesh::elemmap2nodemap(m_elemmap[roll], coor, conn));
347 }
348
349 for (size_t e = 0; e < nconfig; ++e) {
350
351 // Simple shear perturbation
352
353 m_u_s.emplace_back(u.shape());
354 m_Eps_s.emplace_back(Eps.shape());
355 m_Sig_s.emplace_back(Sig.shape());
356
357 xt::xtensor<double, 2> simple_shear = {{0.0, 1.0}, {1.0, 0.0}};
358
359 this->computePerturbation(
360 e, simple_shear, m_u_s[e], m_Eps_s[e], m_Sig_s[e], K, solver, quad, vector, material);
361
362 for (size_t q = 0; q < m_nip; ++q) {
363 auto Epsd = GMatTensor::Cartesian2d::Deviatoric(
364 xt::eval(xt::view(m_Eps_s[e], m_elem_plas(e), q)));
365 double gamma = Epsd(0, 1);
366 for (size_t roll = 0; roll < nroll; ++roll) {
367 auto map = mesh.roll(roll);
368 m_dgamma(e + map(m_elem_plas(e)) - m_elem_plas(e), q) = gamma;
369 }
370 }
371
372 // Pure shear perturbation
373
374 m_u_p.emplace_back(u.shape());
375 m_Eps_p.emplace_back(Eps.shape());
376 m_Sig_p.emplace_back(Sig.shape());
377
378 xt::xtensor<double, 2> pure_shear = {{1.0, 0.0}, {0.0, -1.0}};
379
380 this->computePerturbation(
381 e, pure_shear, m_u_p[e], m_Eps_p[e], m_Sig_p[e], K, solver, quad, vector, material);
382
383 for (size_t q = 0; q < m_nip; ++q) {
384 auto Epsd = GMatTensor::Cartesian2d::Deviatoric(
385 xt::eval(xt::view(m_Eps_p[e], m_elem_plas(e), q)));
386 double E = Epsd(0, 0);
387 for (size_t roll = 0; roll < nroll; ++roll) {
388 auto map = mesh.roll(roll);
389 m_dE(e + map(m_elem_plas(e)) - m_elem_plas(e), q) = E;
390 }
391 }
392 }
393}

395inline void LocalTriggerFineLayerFull::computePerturbation(
396 size_t trigger_plastic,
397 const xt::xtensor<double, 2>& sig_star,
398 xt::xtensor<double, 2>& u,
399 xt::xtensor<double, 4>& Eps,
400 xt::xtensor<double, 4>& Sig,
401 GooseFEM::MatrixPartitioned& K,
402 GooseFEM::MatrixPartitionedSolver<>& solver,
403 const GooseFEM::Element::Quad4::Quadrature& quad,
404 const GooseFEM::VectorPartitioned& vector,
405 GMatElastoPlasticQPot::Cartesian2d::Array<2>& material)
406{
407 size_t trigger_element = m_elem_plas(trigger_plastic);
408
409 Sig.fill(0.0);
410 u.fill(0.0);
411
412 for (size_t q = 0; q < quad.nip(); ++q) {
413 xt::view(Sig, trigger_element, q) = -sig_star;
414 }
415
416 auto fe = quad.Int_gradN_dot_tensor2_dV(Sig);
417 auto fint = vector.AssembleNode(fe);
418 auto fext = xt::zeros_like(fint);
419 vector.copy_p(fint, fext);
420 auto fres = xt::eval(fext - fint);
421
422 solver.solve(K, fres, u);
423
424 vector.asElement(u, fe);
425 quad.symGradN_vector(fe, Eps);
426 material.setStrain(Eps);
427 material.stress(Sig);
428}

496inline void LocalTriggerFineLayerFull::setState(
497 const xt::xtensor<double, 4>& Eps,
498 const xt::xtensor<double, 4>& Sig,
499 const xt::xtensor<double, 2>& epsy,
500 size_t N)
501{
502 FRICTIONQPOTFEM_ASSERT(xt::has_shape(Eps, m_shape_T2));
503 FRICTIONQPOTFEM_ASSERT(xt::has_shape(Sig, m_shape_T2));
504 FRICTIONQPOTFEM_ASSERT(xt::has_shape(epsy, {m_nelem_plas, m_nip}));
505
506 xt::xtensor<double, 2> S = xt::empty<double>({size_t(2), N});
507 xt::xtensor<double, 2> P = xt::empty<double>({size_t(2), N});
508 xt::xtensor<double, 2> W = xt::empty<double>({size_t(2), N});
509
510 for (size_t e = 0; e < m_nelem_plas; ++e) {
511
512 auto Eps_s = this->Eps_s(e);
513 auto Eps_p = this->Eps_p(e);
514 auto Sig_s = this->Sig_s(e);
515 auto Sig_p = this->Sig_p(e);
516 auto Sig_slice = this->slice(Sig, e);
517 auto dV_slice = this->slice(m_dV, e);
518
519 for (size_t q = 0; q < m_nip; ++q) {
520
521 double dgamma = m_dgamma(e, q);
522 double dE = m_dE(e, q);
523
524 auto Epsd =
525 GMatTensor::Cartesian2d::Deviatoric(xt::eval(xt::view(Eps, m_elem_plas(e), q)));
526 double gamma = Epsd(0, 1);
527 double E = Epsd(0, 0);
528 double y = epsy(e, q);
529 double a, b, c, D;
530
531 // solve for "p = 0"
532 a = SQR(dgamma);
533 b = 2.0 * gamma * dgamma;
534 c = SQR(gamma) + SQR(E) - SQR(y);
535 D = SQR(b) - 4.0 * a * c;
536 double smax = (-b + std::sqrt(D)) / (2.0 * a);
537 double smin = (-b - std::sqrt(D)) / (2.0 * a);
538
539 // solve for "s = 0"
540 a = SQR(dE);
541 b = 2.0 * E * dE;
542 c = SQR(E) + SQR(gamma) - SQR(y);
543 D = SQR(b) - 4.0 * a * c;
544 double pmax = (-b + std::sqrt(D)) / (2.0 * a);
545 double pmin = (-b - std::sqrt(D)) / (2.0 * a);
546
547 size_t n = static_cast<size_t>(-smin / (smax - smin) * static_cast<double>(N));
548 size_t m = N - n;
549
550 for (size_t i = 0; i < 2; ++i) {
551 xt::view(S, i, xt::range(0, n)) = xt::linspace<double>(smin, 0, n);
552 xt::view(S, i, xt::range(n, N)) = xt::linspace<double>(smax / double(m), smax, m);
553 P(i, 0) = 0.0;
554 P(i, N - 1) = 0.0;
555 }
556 P(0, n - 1) = pmax;
557 P(1, n - 1) = pmin;
558
559 for (size_t j = 1; j < N - 1; ++j) {
560 if (j == n - 1) {
561 continue;
562 }
563 a = SQR(dE);
564 b = 2.0 * E * dE;
565 c = SQR(E) + std::pow(gamma + S(0, j) * dgamma, 2.0) - SQR(y);
566 D = SQR(b) - 4.0 * a * c;
567 P(0, j) = (-b + std::sqrt(D)) / (2.0 * a);
568 P(1, j) = (-b - std::sqrt(D)) / (2.0 * a);
569 }
570
571 for (size_t i = 0; i < P.shape(0); ++i) {
572 for (size_t j = 0; j < P.shape(1); ++j) {
573 xt::xtensor<double, 4> sig = P(i, j) * Sig_p + S(i, j) * Sig_s + Sig_slice;
574 xt::xtensor<double, 4> deps = P(i, j) * Eps_p + S(i, j) * Eps_s;
575 xt::xtensor<double, 2> w = GMatTensor::Cartesian2d::A2s_ddot_B2s(sig, deps);
576 w *= dV_slice;
577 W(i, j) = xt::sum(w)();
578 }
579 }
580
581 auto idx = xt::argmin(W)();
582 m_smin(e, q) = S.flat(idx);
583 m_pmin(e, q) = P.flat(idx);
584 m_Wmin(e, q) = W.flat(idx);
585 }
586 }

589inline void LocalTriggerFineLayerFull::setStateMinimalSearch(
590 const xt::xtensor<double, 4>& Eps,
591 const xt::xtensor<double, 4>& Sig,
592 const xt::xtensor<double, 2>& epsy)
593{
594 FRICTIONQPOTFEM_ASSERT(xt::has_shape(Eps, m_shape_T2));
595 FRICTIONQPOTFEM_ASSERT(xt::has_shape(Sig, m_shape_T2));
596 FRICTIONQPOTFEM_ASSERT(xt::has_shape(epsy, {m_nelem_plas, m_nip}));
597
598 std::array<double, 8> S;
599 std::array<double, 8> P;
600 std::array<double, 8> W;
601
602 for (size_t e = 0; e < m_nelem_plas; ++e) {
603
604 auto Eps_s = this->Eps_s(e);
605 auto Eps_p = this->Eps_p(e);
606 auto Sig_s = this->Sig_s(e);
607 auto Sig_p = this->Sig_p(e);
608 auto Sig_slice = this->slice(Sig, e);
609 auto dV_slice = this->slice(m_dV, e);
610
611 for (size_t q = 0; q < m_nip; ++q) {
612
613 double dgamma = m_dgamma(e, q);
614 double dE = m_dE(e, q);
615
616 auto Epsd =
617 GMatTensor::Cartesian2d::Deviatoric(xt::eval(xt::view(Eps, m_elem_plas(e), q)));
618 double gamma = Epsd(0, 1);
619 double E = Epsd(0, 0);
620 double y = epsy(e, q);
621 double a, b, c, D;
622
623 // solve for "p = 0"
624 a = SQR(dgamma);
625 b = 2.0 * gamma * dgamma;
626 c = SQR(gamma) + SQR(E) - SQR(y);
627 D = SQR(b) - 4.0 * a * c;
628 double smax = (-b + std::sqrt(D)) / (2.0 * a);
629 double smin = (-b - std::sqrt(D)) / (2.0 * a);
630 P[0] = 0.0;
631 P[1] = 0.0;
632 S[0] = smin;
633 S[1] = smax;
634
635 // solve for "s = 0"
636 a = SQR(dE);
637 b = 2.0 * E * dE;
638 c = SQR(E) + SQR(gamma) - SQR(y);
639 D = SQR(b) - 4.0 * a * c;
640 double pmax = (-b + std::sqrt(D)) / (2.0 * a);
641 double pmin = (-b - std::sqrt(D)) / (2.0 * a);
642 P[2] = pmin;
643 P[3] = pmax;
644 S[2] = 0.0;
645 S[3] = 0.0;
646 S[4] = smin / 2.0;
647 S[5] = smin / 2.0;
648 S[6] = smax / 2.0;
649 S[7] = smax / 2.0;
650
651 for (size_t i = 4; i < S.size(); ++i) {
652 if (i % 2 != 0) {
653 continue;
654 }
655 a = SQR(dE);
656 b = 2.0 * E * dE;
657 c = SQR(E) + std::pow(gamma + S[i] * dgamma, 2.0) - SQR(y);
658 D = SQR(b) - 4.0 * a * c;
659 P[i] = (-b + std::sqrt(D)) / (2.0 * a);
660 P[i + 1] = (-b - std::sqrt(D)) / (2.0 * a);
661 }
662
663 for (size_t i = 0; i < S.size(); ++i) {
664 xt::xtensor<double, 4> sig = P[i] * Sig_p + S[i] * Sig_s + Sig_slice;
665 xt::xtensor<double, 4> deps = P[i] * Eps_p + S[i] * Eps_s;
666 xt::xtensor<double, 2> w = GMatTensor::Cartesian2d::A2s_ddot_B2s(sig, deps);
667 w *= dV_slice;
668 W[i] = xt::sum(w)();
669 }
670
671 auto idx = std::distance(W.begin(), std::min_element(W.begin(), W.end()));
672 m_smin(e, q) = S[idx];
673 m_pmin(e, q) = P[idx];
674 m_Wmin(e, q) = W[idx];
675 }
676 }

689 std::array<double, 2> W;
690

692
693 auto Eps_s = this->Eps_s(e);
694 auto Sig_s = this->Sig_s(e);

713 D = SQR(b) - 4.0 * a * c;

727 idx = 1;

730 m_smin(e, q) = S[idx];
731 m_pmin(e, q) = 0.0;
