375 template <class T>
376 void rshift_y(long istart, const T& y, size_t nbuffer = 0)
377 {
378 QPOT_ASSERT(y.size() > 0);
379 QPOT_ASSERT(istart <= m_istop);
380 QPOT_ASSERT(istart >= m_istart); // otherwise the function does not what it's name suggests
381
382// if there is overlap between the chunks of Y in memory and input:
383// check that the values are indeed the same
384#ifdef QPOT_ENABLE_DEBUG
385 if (istart < m_istop) {
386 size_t d = static_cast<size_t>(m_istop - istart);
387 xt::xtensor<double, 1> o = xt::empty<double>({d});
388 xt::xtensor<double, 1> n = xt::empty<double>({d});
389 std::copy(m_y.cend() - d, m_y.cend(), o.begin());
390 std::copy(y.cbegin(), y.cbegin() + d, n.begin());
391 QPOT_DEBUG(xt::allclose(o, n));
392 }
393#endif
394
395 // supplied chunk smaller than the chunk in memory
396 if (istart >= m_istart && istart + static_cast<long>(y.size()) <= m_istop) {
397 return;
398 }
399
400 m_istart = m_istop - static_cast<long>(nbuffer);
401
402 // take everything for "y": it spans the new entries and the buffered old entries
403 if (istart <= m_istart) {
404 return this->set_y(istart, y);
405 }
406
407 // buffer "m_y[m_n - nbuffer:]" and take the rest of "y"
408 m_ymin_data = *y.cbegin();
409 QPOT_ASSERT(nbuffer <= m_n);
410 QPOT_DEBUG(std::is_sorted(y.cbegin(), y.cend()));
411 size_t d = static_cast<size_t>(m_istop - istart); // items to skip from "y"
412 m_istop = istart + static_cast<long>(y.size());
413 m_n = static_cast<size_t>(m_istop - m_istart);
414 QPOT_ASSERT(m_n >= nbuffer);
415
416 std::copy(m_y.cend() - nbuffer, m_y.cend(), m_y.begin());
417 m_y.resize(m_n);
418 std::copy(y.cbegin() + d, y.cend(), m_y.begin() + nbuffer);
419
420 m_full_search = true;
421 this->update();
422 }

807private:
808 void update()
809 {
810 if (m_x <= m_y.front()) {
811 m_redraw = -1;
812 }
813 else if (m_x > m_y.back()) {
814 m_redraw = +1;
815 }
816 else {
817 m_redraw = 0;
818 }
819
820 m_lock = m_redraw != 0;
821
822 if (m_lock) {
823 return;
824 }
825
826 if (m_full_search) {
827 m_li = std::lower_bound(m_y.cbegin(), m_y.cend(), m_x) - m_y.cbegin() - 1;
828 m_left = m_y[m_li];
829 m_right = m_y[m_li + 1];
830 m_full_search = false;
831 return;
832 }
833
834 if (m_left < m_x && m_right >= m_x) {
835 return;
836 }
837
838 size_t l = m_li > m_proximity ? m_li - m_proximity : 0;
839 size_t r = std::min(m_li + m_proximity, m_n - 1);
840
841 if (m_y[l] < m_x && m_y[r] >= m_x) {
842 m_li = std::lower_bound(&m_y[l], &m_y[l] + r - l, m_x) - &m_y[l] - 1 + l;
843 }
844 else {
845 m_li = std::lower_bound(m_y.cbegin(), m_y.cend(), m_x) - m_y.cbegin() - 1;
846 }
847
848 m_left = m_y[m_li];
849 m_right = m_y[m_li + 1];
850 }
