31inline double Elastic::energy() const
32{
33 namespace GT = GMatTensor::Cartesian2d::pointer;
34 std::array<double, 4> Epsd;
35 double epsm = GT::Hydrostatic_deviatoric(&m_Eps[0], &Epsd[0]);
36 double epsd = std::sqrt(0.5 * GT::A2s_ddot_B2s(&Epsd[0], &Epsd[0]));
37 double U = m_K * std::pow(epsm, 2.0);
38 double V = m_G * std::pow(epsd, 2.0);

44{
