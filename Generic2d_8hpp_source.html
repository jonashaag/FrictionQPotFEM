604inline double System::eventDrivenStep(
605 double deps_kick,
606 bool kick,
607 int direction,
608 bool yield_element,
609 bool iterative)
610{
611 FRICTIONQPOTFEM_ASSERT(xt::has_shape(m_pert_u, m_u.shape()));
612 FRICTIONQPOTFEM_ASSERT(direction == 1 || direction == -1);
613
614 auto eps = GMatElastoPlasticQPot::Cartesian2d::Epsd(this->plastic_Eps());
615 auto idx = this->plastic_CurrentIndex();
616 auto epsy_l = this->plastic_CurrentYieldLeft();
617 auto epsy_r = this->plastic_CurrentYieldRight();
618
619 FRICTIONQPOTFEM_WIP(iterative || direction > 0 || !xt::any(xt::equal(idx, 0)));
620
621 xt::xtensor<double, 2> target;
622
623 if (direction > 0 && kick) { // direction > 0 && kick
624 target = epsy_r + 0.5 * deps_kick;
625 }
626 else if (direction > 0) { // direction > 0 && !kick
627 target = epsy_r - 0.5 * deps_kick;
628 }
629 else if (kick) { // direction < 0 && kick
630 target = epsy_l - 0.5 * deps_kick;
631 }
632 else { // direction < 0 && !kick
633 target = epsy_l + 0.5 * deps_kick;
634 }
635
636 auto Epsd_plastic = GMatElastoPlasticQPot::Cartesian2d::Deviatoric(this->plastic_Eps());
637 auto scale = xt::empty_like(target);
638
639 for (size_t e = 0; e < m_nelem_plas; ++e) {
640 for (size_t q = 0; q < m_nip; ++q) {
641
642 double e_t = Epsd_plastic(e, q, 0, 0);
643 double g_t = Epsd_plastic(e, q, 0, 1);
644 double e_d = m_pert_Epsd_plastic(e, q, 0, 0);
645 double g_d = m_pert_Epsd_plastic(e, q, 0, 1);
646 double epsd_target = target(e, q);
647
648 double a = e_d * e_d + g_d * g_d;
649 double b = 2.0 * (e_t * e_d + g_t * g_d);
650 double c = e_t * e_t + g_t * g_t - epsd_target * epsd_target;
651 double D = std::sqrt(b * b - 4.0 * a * c);
652
653 FRICTIONQPOTFEM_REQUIRE(b >= 0.0);
654 scale(e, q) = (-b + D) / (2.0 * a);
655 }
656 }
657
658 double ret;
659 size_t e;
660 size_t q;
661
662 if (!iterative) {
663
664 auto index = xt::unravel_index(xt::argmin(xt::abs(scale))(), scale.shape());
665 e = index[0];
666 q = index[1];
667
668 if (kick && yield_element) {
669 q = xt::argmax(xt::view(xt::abs(scale), e, xt::all()))();
670 }
671
672 ret = scale(e, q);
673
674 if ((direction > 0 && ret < 0) || (direction < 0 && ret > 0)) {
675 if (!kick) {
676 return 0.0;
677 }
678 else {
679 FRICTIONQPOTFEM_REQUIRE((direction > 0 && ret > 0) || (direction < 0 && ret < 0));
680 }
681 }
682 }
683
684 else {
685
686 double dir = static_cast<double>(direction);
687 auto steps = xt::sort(xt::ravel(xt::eval(xt::abs(scale))));
688 auto u_n = this->u();
689 auto jdx = xt::cast<long>(this->plastic_CurrentIndex());
690 size_t i;
691 long nip = static_cast<long>(m_nip);
692
693 // find first step that:
694 // if (!kick || (kick && !yield_element)): is plastic for the first time
695 // if (kick && yield_element): yields the element for the first time
696
697 for (i = 0; i < steps.size(); ++i) {
698 this->setU(u_n + dir * steps(i) * m_pert_u);
699 auto jdx_new = xt::cast<long>(this->plastic_CurrentIndex());
700 auto S = xt::abs(jdx_new - jdx);
701 if (!yield_element || !kick) {
702 if (xt::any(S > 0)) {
703 break;
704 }
705 }
706 else {
707 if (xt::any(xt::equal(xt::sum(S > 0, 1), nip))) {
708 break;
709 }
710 }
711 }
712
713 // iterate to acceptable step
714
715 double right = steps(i);
716 double left = 0.0;
717 ret = right;
718
719 if (i > 0) {
720 left = steps(i - 1);
721 }
722
723 // iterate to actual step
724
725 for (size_t iiter = 0; iiter < 1100; ++iiter) {
726
727 ret = 0.5 * (right + left);
728 this->setU(u_n + dir * ret * m_pert_u);
729 auto jdx_new = xt::cast<long>(this->plastic_CurrentIndex());
730 auto S = xt::abs(jdx_new - jdx);
731
732 if (!kick) {
733 if (xt::any(S > 0)) {
734 right = ret;
735 }
736 else {
737 left = ret;
738 }
739 }
740 else if (yield_element) {
741 if (xt::any(xt::equal(xt::sum(S > 0, 1), nip))) {
742 right = ret;
743 }
744 else {
745 left = ret;
746 }
747 }
748 else {
749 if (xt::any(S > 0)) {
750 right = ret;
751 }
752 else {
753 left = ret;
754 }
755 }
756
757 if ((right - left) / left < 1e-5) {
758 break;
759 }
760 FRICTIONQPOTFEM_REQUIRE(iiter < 1000);
761 }
762
763 // final assertion: make sure that "left" and "right" are still bounds
764
765 {
766 this->setU(u_n + dir * left * m_pert_u);
767 auto jdx_new = xt::cast<long>(this->plastic_CurrentIndex());
768 auto S = xt::abs(jdx_new - jdx);
769
770 FRICTIONQPOTFEM_REQUIRE(kick || xt::all(xt::equal(S, 0)));
771 if (kick && yield_element) {
772 FRICTIONQPOTFEM_REQUIRE(!xt::any(xt::equal(xt::sum(S > 0, 1), nip)));
773 }
774 else if (kick) {
775 FRICTIONQPOTFEM_REQUIRE(xt::all(xt::equal(S, 0)));
776 }
777 }
778 {
779 this->setU(u_n + dir * right * m_pert_u);
780 auto jdx_new = xt::cast<long>(this->plastic_CurrentIndex());
781 auto S = xt::abs(jdx_new - jdx);
782 FRICTIONQPOTFEM_REQUIRE(!xt::all(xt::equal(S, 0)));
783
784 FRICTIONQPOTFEM_REQUIRE(kick || !xt::all(xt::equal(S, 0)));
785 if (kick && yield_element) {
786 FRICTIONQPOTFEM_REQUIRE(xt::any(xt::equal(xt::sum(S > 0, 1), nip)));
787 }
788 else if (kick) {
789 FRICTIONQPOTFEM_REQUIRE(xt::any(S > 0));
790 }
791 }
792
793 // "output"
794
795 if (!kick) {
796 ret = dir * left;
797 }
798 else {
799 ret = dir * right;
800 }
801 this->setU(u_n);
802 FRICTIONQPOTFEM_REQUIRE((direction > 0 && ret >= 0) || (direction < 0 && ret <= 0));
803 }
804
805 this->setU(m_u + ret * m_pert_u);
806
807 auto idx_new = this->plastic_CurrentIndex();
808 FRICTIONQPOTFEM_REQUIRE(kick || xt::all(xt::equal(idx, idx_new)));
809 FRICTIONQPOTFEM_REQUIRE(!kick || xt::any(xt::not_equal(idx, idx_new)));
810
811 if (!iterative) {
812 auto eps_new = GMatElastoPlasticQPot::Cartesian2d::Epsd(this->plastic_Eps())(e, q);
813 auto eps_target = target(e, q);
814 FRICTIONQPOTFEM_REQUIRE(xt::allclose(eps_new, eps_target));
815 }
816
817 return ret;
818}

820inline void System::timeStep()
821{
822 FRICTIONQPOTFEM_ASSERT(m_allset);
823
824 // history
825
826 m_t += m_dt;
827 xt::noalias(m_v_n) = m_v;
828 xt::noalias(m_a_n) = m_a;
829
830 // new displacement
831
832 xt::noalias(m_u) = m_u + m_dt * m_v + 0.5 * std::pow(m_dt, 2.0) * m_a;
833 this->updated_u();
834
835 // estimate new velocity, update corresponding force
836
837 xt::noalias(m_v) = m_v_n + m_dt * m_a_n;
838 this->updated_v();
839
840 // compute residual force & solve
841
842 this->computeInternalExternalResidualForce();
843 m_M.solve(m_fres, m_a);
844
845 // re-estimate new velocity, update corresponding force
846
847 xt::noalias(m_v) = m_v_n + 0.5 * m_dt * (m_a_n + m_a);
848 this->updated_v();
849
850 // compute residual force & solve
851
852 this->computeInternalExternalResidualForce();
853 m_M.solve(m_fres, m_a);
854
855 // new velocity, update corresponding force
856
857 xt::noalias(m_v) = m_v_n + 0.5 * m_dt * (m_a_n + m_a);
858 this->updated_v();
859
860 // compute residual force & solve
861
862 this->computeInternalExternalResidualForce();
863 m_M.solve(m_fres, m_a);
864}
