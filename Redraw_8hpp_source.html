331inline RedrawList::RedrawList(const E& x, F draw, size_t ntotal, size_t nbuffer, size_t noffset)
332 : m_N(x.size()), m_ntot(ntotal), m_nbuf(nbuffer), m_noff(noffset), m_draw(draw)
333{
334 QPOT_ASSERT(x.dimension() == 1);
335 QPOT_ASSERT(m_nbuf <= m_ntot);
336 QPOT_ASSERT(m_noff <= m_nbuf);
337 QPOT_ASSERT(2 * m_noff < m_ntot); // either redraw left or right, never both for the same "x"
338
339 m_proximity = std::min(m_proximity, m_ntot);
340
341 // draw yield distance (times two!), and convert to yield positions
342 m_val = m_draw({m_N, m_ntot});
343 m_pos = xt::cumsum(m_val, 1);
344
345 // shift such that the mean of "x" is in the middle of the potential energy landscape
346 auto shift = xt::mean(m_pos, 1) - xt::mean(x);
347 for (size_t p = 0; p < m_N; ++p) {
348 xt::view(m_pos, p, xt::all()) -= shift(p);
349 }
350
351 // register minimum and maximum positions for each particle
352 m_min = xt::view(m_pos, xt::all(), 0);
353 m_max = xt::view(m_pos, xt::all(), m_pos.shape(1) - 1);
354 QPOT_ASSERT(xt::all(x > m_min));
355 QPOT_ASSERT(xt::all(x < m_max));
356
357 // allocate current index
358 m_idx = xt::empty<decltype(m_idx)::value_type>({m_N});
359 m_idx_t = xt::zeros<decltype(m_idx_t)::value_type>({m_N});
360 m_iredraw = xt::zeros<decltype(m_iredraw)::value_type>({m_N});
361 m_left = xt::empty<decltype(m_left)::value_type>({m_N});
362 m_right = xt::empty<decltype(m_right)::value_type>({m_N});
363
364 // compute current index (updates "m_idx" only, "m_idx_t" is only changed at redraw)
365 for (size_t p = 0; p < m_N; ++p) {
366 size_t i =
367 std::lower_bound(&m_pos(p, 0), &m_pos(p, 0) + m_pos.shape(1), x(p)) - &m_pos(p, 0) - 1;
368 m_idx(p) = i;
369 m_left(p) = m_pos(p, i);
370 m_right(p) = m_pos(p, i + 1);
371 }
372}

422inline xt::xtensor<double, 1> RedrawList::nextYield(int offset) const
423{
424 QPOT_ASSERT(!m_lock);
425 QPOT_ASSERT(offset != 0);
426
427 if (offset < 0) {

488template <class T>
489inline void RedrawList::redrawRight(const T& index)
490{
491 // store redraw
492 m_lock = true;
493 m_redraw = true;
494 xt::view(m_iredraw, xt::keep(index)) = +1;
495
496 // allocate new yield distances (times two!)
497 xt::xtensor<double, 2> y = xt::empty<double>({index.size(), m_ntot});
498
499 // apply buffer: insert previously drawn values
500 xt::view(y, xt::all(), xt::range(0, m_nbuf)) =
501 xt::view(m_val, xt::keep(index), xt::range(m_ntot - m_nbuf, m_ntot));
502
503 // draw yield distances (times two!)
504 xt::view(y, xt::all(), xt::range(m_nbuf, m_ntot)) = m_draw({index.size(), m_ntot - m_nbuf});
505 QPOT_ASSERT(xt::all(y > 0));
506
507 // store yield distances
508 xt::view(m_val, xt::keep(index), xt::all()) = y;
509
510 // update yield positions
511 xt::view(y, xt::all(), 0) = xt::view(m_pos, xt::keep(index), m_ntot - m_nbuf);
512 xt::view(m_pos, xt::keep(index), xt::all()) = xt::cumsum(y, 1);
513
514 // register minimum and maximum yield positions for each particle
515 xt::noalias(m_min) = xt::view(m_pos, xt::all(), 0);
516 xt::noalias(m_max) = xt::view(m_pos, xt::all(), m_pos.shape(1) - 1);
517
518 // update historic/current index
519 xt::view(m_idx_t, xt::keep(index)) += m_ntot - m_nbuf;
520}

536
537 // draw yield distances (times two!)

539 QPOT_ASSERT(xt::all(y > 0));

556template <class E>
557inline void RedrawList::computeIndex(const E& x)
558{
559 using index_type = decltype(m_idx)::value_type;
560 using position_type = typename E::value_type;
561
562 if (!m_redraw) {
563 for (size_t p = 0; p < m_N; ++p) {
564 index_type i = m_idx(p);
565 position_type xp = x(p);
566
567 if (m_left(p) < xp && m_right(p) >= xp) {
568 continue;
569 }
570
571 index_type l = i > m_proximity ? i - m_proximity : 0;
572 index_type r = std::min(i + m_proximity, m_ntot - 1);
573
574 if (m_pos(p, l) < xp && m_pos(p, r) >= xp) {
575 i = std::lower_bound(&m_pos(p, l), &m_pos(p, l) + r - l, xp) - &m_pos(p, l) - 1 + l;
576 }
577 else {
578 i = std::lower_bound(&m_pos(p, 0), &m_pos(p, 0) + m_ntot, xp) - &m_pos(p, 0) - 1;
579 }
580
581 m_idx(p) = i;
582 m_left(p) = m_pos(p, i);
583 m_right(p) = m_pos(p, i + 1);
584 }
585 }
586 else {
587 for (size_t p = 0; p < m_N; ++p) {
588 index_type i = m_idx(p);
589 position_type xp = x(p);
590 bool redraw = m_iredraw(p) != 0;
591
592 if (m_left(p) < xp && m_right(p) >= xp && !redraw) {
593 continue;
594 }
595
596 index_type l = i > m_proximity ? i - m_proximity : 0;
597 index_type r = std::min(i + m_proximity, m_ntot - 1);
598
599 if (m_pos(p, l) < xp && m_pos(p, r) >= xp && !redraw) {
600 i = std::lower_bound(&m_pos(p, l), &m_pos(p, l) + r - l, xp) - &m_pos(p, l) - 1 + l;
601 }
602 else {
603 i = std::lower_bound(&m_pos(p, 0), &m_pos(p, 0) + m_ntot, xp) - &m_pos(p, 0) - 1;
604 }
605
606 m_idx(p) = i;
607 m_left(p) = m_pos(p, i);
608 m_right(p) = m_pos(p, i + 1);
609 }
610 }
611
612 m_lock = false;
613}

639template <class E>
640inline bool RedrawList::setPosition(const E& x)
641{
642 QPOT_ASSERT(x.size() == m_N);
643
644 // if redraw was force-triggered (from redrawRight or redrawLeft and not from here):
645 // make sure that all indices are updated (forces update for 'redrawn particles')
646 if (m_lock) {
647 this->computeIndex(x);
648 }
649
650 // erase history
651 if (m_redraw) {
652 m_iredraw.fill(0);
653 m_redraw = false;
654 }
655
656 // extend right (use the opportunity to redraw multiple rows)
657 if (xt::any(x >= m_max)) {
658 xt::xtensor<size_t, 1> index =
659 xt::flatten_indices(xt::argwhere(x >= xt::view(m_pos, xt::all(), m_ntot - m_noff)));
660
661 this->redrawRight(index);
662 }
663
664 // extend left (use the opportunity to redraw multiple rows)
665 if (xt::any(x <= m_min)) {
666 // get rows for which a redraw is needed
667 // use the opportunity to redraw multiple rows
668 xt::xtensor<size_t, 1> index =
669 xt::flatten_indices(xt::argwhere(x <= xt::view(m_pos, xt::all(), m_noff)));
670
671 this->redrawLeft(index);
672 }
673
674 // update index (force update 'redrawn particles')
675 this->computeIndex(x);
676
677 return m_redraw;
678}
