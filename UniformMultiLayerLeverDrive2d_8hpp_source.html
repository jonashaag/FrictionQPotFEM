92template <class T>
93inline void System::initEventDriven(double xlever, const T& active)
94{
95 FRICTIONQPOTFEM_ASSERT(this->isHomogeneousElastic());
96 FRICTIONQPOTFEM_ASSERT(xt::has_shape(active, m_layerdrive_active.shape()));
97
98 // backup system
99
100 auto u0 = m_u;
101 auto v0 = m_v;
102 auto a0 = m_a;
103 auto active0 = m_layerdrive_active;
104 auto ubar0 = m_layerdrive_targetubar;
105 auto xdrive0 = m_lever_target;
106 auto t0 = m_t;
107 xt::xtensor<double, 3> epsy0(std::array<size_t, 3>{m_nelem_plas, m_nip, 2});
108
109 for (size_t e = 0; e < m_nelem_plas; ++e) {
110 for (size_t q = 0; q < m_nip; ++q) {
111 auto& cusp = m_material_plas.refCusp({e, q});
112 auto epsy = cusp.epsy();
113 epsy0(e, q, 0) = epsy(0);
114 epsy0(e, q, 1) = epsy(1);
115 epsy(1) = std::numeric_limits<double>::max();
116 epsy(0) = -epsy(1);
117 cusp.reset_epsy(epsy, false);
118 }
119 }
120
121 // perturbation
122
123 m_u.fill(0.0);
124 m_v.fill(0.0);
125 m_a.fill(0.0);
126 this->updated_u();
127 FRICTIONQPOTFEM_ASSERT(xt::all(xt::equal(this->plastic_CurrentIndex(), 0)));
128 this->layerSetTargetActive(active);
129 this->setLeverTarget(xlever);
130 this->minimise();
131 FRICTIONQPOTFEM_ASSERT(xt::all(xt::equal(this->plastic_CurrentIndex(), 0)));
132
133 auto c = this->eventDriven_setDeltaU(m_u);
134 m_pert_layerdrive_active = active;
135 m_pert_layerdrive_targetubar = c * m_layerdrive_targetubar;
136 m_pert_lever_target = c * xlever;
137
138 // restore system
139
140 for (size_t e = 0; e < m_nelem_plas; ++e) {
141 for (size_t q = 0; q < m_nip; ++q) {
142 auto& cusp = m_material_plas.refCusp({e, q});
143 auto epsy = cusp.epsy();
144 epsy(0) = epsy0(e, q, 0);
145 epsy(1) = epsy0(e, q, 1);
146 cusp.reset_epsy(epsy, false);
147 }
148 }
149
150 this->setU(u0);
151 this->setV(v0);
152 this->setA(a0);
153 this->layerSetTargetActive(active0);
154 this->layerSetTargetUbar(ubar0);
155 this->setLeverTarget(xdrive0);
156 this->setT(t0);
157}

186inline void System::updated_u()
187{
188 FRICTIONQPOTFEM_ASSERT(m_lever_set);
189
190 this->computeLayerUbarActive();
191
192 // Position of the lever based on equilibrium
193
194 m_lever_u = m_lever_target;
195 double n = 1.0;
196
197 for (size_t i = 0; i < m_n_layer; ++i) {
198 if (m_layerdrive_active(i, 0)) {
199 m_lever_u += m_layer_ubar(i, 0) * m_lever_hi_H(i);
200 n += m_lever_hi_H_2(i);
201 }
202 }
203
204 m_lever_u /= n;
205
206 // Update position of driving springs
207
208 for (size_t i = 0; i < m_n_layer; ++i) {
209 if (m_layerdrive_active(i, 0)) {
210 m_layerdrive_targetubar(i, 0) = m_lever_hi_H(i) * m_lever_u;
211 }
212 }
213
214 // Update forces
215 this->computeForceFromTargetUbar();
216 this->computeForceMaterial();
217}
