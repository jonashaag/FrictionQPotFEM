17inline Vector::Vector(const S& conn, const T& dofs) : m_conn(conn), m_dofs(dofs)
18{
19 GOOSEFEM_ASSERT(conn.dimension() == 2);
20 GOOSEFEM_ASSERT(dofs.dimension() == 2);
21
22 m_nelem = m_conn.shape(0);
23 m_nne = m_conn.shape(1);
24 m_nnode = m_dofs.shape(0);
25 m_ndim = m_dofs.shape(1);
26 m_ndof = xt::amax(m_dofs)() + 1;
27
28 GOOSEFEM_ASSERT(xt::amax(m_conn)() + 1 <= m_nnode);
29 GOOSEFEM_ASSERT(m_ndof <= m_nnode * m_ndim);
30}

131inline void Vector::asDofs_impl_nodevec(const T& arg, R& dofval) const
132{
133 static_assert(
134 xt::get_rank<R>::value == 1 || !xt::has_fixed_rank_t<R>::value, "Unknown rank 'ret'");
135 GOOSEFEM_ASSERT(xt::has_shape(arg, this->shape_nodevec()));
136 GOOSEFEM_ASSERT(xt::has_shape(dofval, this->shape_dofval()));

138 dofval.fill(0.0);
139
140#pragma omp parallel for
141 for (size_t m = 0; m < m_nnode; ++m) {
142 for (size_t i = 0; i < m_ndim; ++i) {
143 dofval(m_dofs(m, i)) = arg(m, i);
